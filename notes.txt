1. install package first & change package.json: 

- npx express-generator -e
- npm i nodemon
- "scripts": {
    "start": "nodemon ./bin/www"
  },

2. connect mongoose: 
    const mongoose = require('mongoose')
    mongoose.connect('mongodb://localhost:27017/admin')
    .then(() => {
        console.log('connected successfully')
    })
    .catch((error) => {
        console.log(error)
    })


3. create ejs file:
- in this file have crud structure(code of your crud)

4. create controller & models : 
- file name should be same for both 

    controller : 
    - create logic of crud : viewPage, createPage, editPage, deletePage

    models : 
    const mongoose = require('mongoose')
    const userSchema = new mongoose.Schema({
        name: String,
        age: Number
    })
    module.exports = mongoose.model('collection name what you want', userSchema)

5. call & run :
- call the controller file you created & store it in variable :
    let UC = require('./controller/user')

- call them in app.js:
    app.get('/', UC.viewPage)
    app.get('/createData', UC.createData)
    app.get('/deleteData/:deleteId', UC.deleteData)
    app.get('/editData/:editId', UC.editData)    

extra : 
- routes of file if need :
    var express = require('express');
    var router = express.Router();
    const USER = require('../models/user');

    /* GET users listing. */
    router.get('/', async (req, res) => {

        const data = await USER.find()
  
        res.render('user.ejs', { user : data });
    });

    module.exports = router;
